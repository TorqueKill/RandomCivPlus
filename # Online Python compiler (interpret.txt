# Define the string with the data
text = st

# Split the text string into lines
lines = text.split('\n')

# Initialize an empty array to store the objects
objects = []

# Define the scores dictionary
scores = {1: 8, 2: 5, 3: 4, 4: 3, 5: 2, 6: 1}
_id=1
# Iterate over each line of the text
for line in lines:
    # Split the line into wonder name and score
    parts = line.split('\t')
    
    # Extract the wonder name and score
    wonder_name = parts[0]
    wonder_score = scores[int(parts[1])]
    
    # Create the object and add it to the array
    object = {'id': _id,'name': wonder_name, 'score': wonder_score}
    _id+=1
    
    objects.append(object)

# Print the object array
for i in objects:
    print(i)
